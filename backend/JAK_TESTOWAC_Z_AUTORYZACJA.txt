Flow w Swaggerze:

- Zarejestruj się: POST /api/auth/register/client

{
"email":"test@example.com",
"password":"Test123!",
"firstName":"Jan",
"lastName":"Kowalski"

- Zaloguj: POST /api/auth/login -> skopiuj accessToken
- Kliknij "Authorize" w prawym górnym rogu Swaggera
- Wpisz: Bearer TWOJ_TOKEN -> "Authorize" -> "Close"
- Teraz mozna testować: GET/PUT /api/client/profile

    "Content-Type" = "application/json"
}

    firstName = "Janusz"
} | ConvertTo-Json

3. Wykonaj zapytanie
try {
    $response = Invoke-RestMethod -Uri "https://localhost:7026/api/Client/profile" -Method PUT -Headers $headers -Body $body
    Write-Host "SUKCES:" -ForegroundColor Green
    $response | ConvertTo-Json
}
catch {
    Write-Host "BŁĄD:" -ForegroundColor Red
    Write-Host "Status Code: $($_.Exception.Response.StatusCode.value__)"
    Write-Host "Message: $($_.Exception.Message)"
    
    # Spróbuj odczytać odpowiedź błędu
    $reader = New-Object System.IO.StreamReader($_.Exception.Response.GetResponseStream())
    $reader.BaseStream.Position = 0
    $reader.DiscardBufferedData()
    $errorBody = $reader.ReadToEnd()
    Write-Host "Response Body: $errorBody"
}
